a = 0
b = 0

for(i in 1:2000){


x = seq(20,80,by = 0.01)
n = length(x)
err = rnorm(n,0,5)
y = 1+1.2*x+err

fit = lm(y~x) # ~: alt+126
a[i] = fit$coeff[1]
b[i] = fit$coeff[2]

abline(a[i],b[i],col=5)

}
