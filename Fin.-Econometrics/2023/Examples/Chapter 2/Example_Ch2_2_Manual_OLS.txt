
#EXAMPLE, MANUALLY COMPUTE OLS

x = 1:1:100
n = length(x)
err = rnorm(n,0,2)
y = 1+1.2*x+err
plot(x,y)

xbar = mean(x)
ybar = mean(y)
xy = sum(x*y)

beta_hat = (xy-n*xbar*ybar)/(sum(x^2)-n*xbar^2)
alpha_hat = ybar - beta_hat*xbar

abline(alpha_hat,beta_hat)

summary(lm(y~x))
